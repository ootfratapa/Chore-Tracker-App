        <form class="chore-form" id="choreForm">
            <input type="text" id="choreInput" placeholder="New task..." required autocomplete="off">
            <input type="number" id="coinValueInput" value="1" min="1" required title="Coin Reward">
            <button type="submit" class="add-btn">Add</button>
        </form>

        <ul class="chore-list" id="choreList"></ul>
    </div>

    <script>
        const choreForm = document.getElementById('choreForm');
        const choreInput = document.getElementById('choreInput');
        const coinValueInput = document.getElementById('coinValueInput');
        const choreList = document.getElementById('choreList');
        const coinCountDisplay = document.getElementById('coinCount');

        let chores = JSON.parse(localStorage.getItem('chores')) || [];
        let totalCoins = parseInt(localStorage.getItem('totalCoins')) || 0;

        function saveAndRender() {
            localStorage.setItem('chores', JSON.stringify(chores));
            localStorage.setItem('totalCoins', totalCoins.toString());
            render();
        }

        function render() {
            choreList.innerHTML = '';
            chores.forEach(chore => {
                const li = document.createElement('li');
                li.className = `chore-item ${chore.completed ? 'completed' : ''}`;
                li.innerHTML = `
                    <div class="chore-left" onclick="toggleChore(${chore.id})">
                        <div class="checkbox"></div>
                        <div>
                            <div class="chore-title">${chore.text}</div>
                            <div class="chore-coin-reward">🪙 +${chore.coins}</div>
                        </div>
                    </div>
                    <button class="delete-btn" onclick="deleteChore(${chore.id})">×</button>
                `;
                choreList.appendChild(li);
            });
            coinCountDisplay.textContent = totalCoins;
        }

        choreForm.addEventListener('submit', (e) => {
            e.preventDefault();
            chores.push({
                id: Date.now(),
                text: choreInput.value.trim(),
                coins: parseInt(coinValueInput.value) || 1,
                completed: false
            });
            choreInput.value = '';
            coinValueInput.value = '1';
            saveAndRender();
        });

        window.toggleChore = function(id) {
            chores = chores.map(chore => {
                if (chore.id === id) {
                    const isNowCompleted = !chore.completed;
                    totalCoins += isNowCompleted ? chore.coins : -chore.coins;
                    return { ...chore, completed: isNowCompleted };
                }
                return chore;
            });
            saveAndRender();
        };

        window.deleteChore = function(id) {
            chores = chores.filter(chore => chore.id !== id);
            saveAndRender();
        };

        window.resetCoins = function() {
            if (confirm("Reset total coins to 0?")) {
                totalCoins = 0;
                saveAndRender();
            }
        };

        render();
    </script>
